const SHEET_ID = '1RHRei2j1xwUoXqubuKmHj3jgz86gJfgETzDt3hQNFU8';
const SHEET_NAME = 'Notifications';

// Initialize sheet if doesn't exist
function initializeSheet() {
  const ss = SpreadsheetApp.openById(SHEET_ID);
  let sheet = ss.getSheetByName(SHEET_NAME);

  if (!sheet) {
    // Sheet doesn't exist, create it.
    sheet = ss.insertSheet(SHEET_NAME);

    // Add headers
    sheet.appendRow(['Timestamp', 'PNR', 'Title', 'Body', 'Language', 'Status']);
  }

  return sheet;
}

function doPost(e) {
  try {
    const data = JSON.parse(e.postData.contents || '{}');
    const action = (e.parameter && e.parameter.action) || data.action;
    
    if (action === 'save') {
      initializeSheet();
      saveNotification(data);
      return ContentService.createTextOutput(JSON.stringify({
        success: true,
        message: 'Notification saved'
      })).setMimeType(ContentService.MimeType.JSON);
    }
    
  } catch(error) {
    return ContentService.createTextOutput(JSON.stringify({
      success: false,
      error: error.toString()
    })).setMimeType(ContentService.MimeType.JSON);
  }
}

function doGet(e) {
  try {
    const action = e.parameter.action;
    const pnr = e.parameter.pnr;
    
    initializeSheet();
    
    if (action === 'getByPNR' && pnr) {
      const notifications = getNotificationsByPNR(pnr);
      return ContentService.createTextOutput(JSON.stringify({
        success: true,
        notifications: notifications
      })).setMimeType(ContentService.MimeType.JSON);
    }
    
    if (action === 'getAll') {
      const notifications = getAllNotifications();
      return ContentService.createTextOutput(JSON.stringify({
        success: true,
        data: notifications
      })).setMimeType(ContentService.MimeType.JSON);
    }
    
  } catch(error) {
    return ContentService.createTextOutput(JSON.stringify({
      success: false,
      error: error.toString()
    })).setMimeType(ContentService.MimeType.JSON);
  }
  
  return ContentService.createTextOutput(JSON.stringify({
    success: false,
    error: 'Invalid action'
  })).setMimeType(ContentService.MimeType.JSON);
}

function saveNotification(data) {
  const sheet = initializeSheet();
  const timestamp = new Date().toLocaleString('vi-VN');
  
  sheet.appendRow([
    timestamp,
    data.pnr || '',
    data.title || '',
    data.body || '',
    data.lang || 'vi',
    data.sendImmediately ? 'sent' : 'pending'
  ]);
}

function getNotificationsByPNR(pnr) {
  const sheet = initializeSheet();
  const data = sheet.getDataRange().getValues();
  const result = [];
  const targetPnr = String(pnr || '').toUpperCase();
  
  for (let i = 1; i < data.length; i++) {
    if (String(data[i][1] || '').toUpperCase() === targetPnr) {
      result.push({
        timestamp: data[i][0],
        title: data[i][2],
        body: data[i][3],
        lang: data[i][4],
        status: data[i][5]
      });
    }
  }
  
  return result;
}

function getAllNotifications() {
  const sheet = initializeSheet();
  const data = sheet.getDataRange().getValues();
  const result = [];
  
  for (let i = 1; i < data.length; i++) {
    result.push({
      timestamp: data[i][0],
      pnr: data[i][1],
      title: data[i][2],
      body: data[i][3],
      lang: data[i][4],
      status: data[i][5]
    });
  }
  
  return result;
}
